angular.module('app', ['pascalprecht.translate']) .run(['session', '$log', '$translate', function (session, $log, $translate) { // get users preferred language from session session.getData('language', function (error, result) { if (error) { $log.error(error); } // use language if (result && result.language) { $translate.uses(result.language); $log.info('Language key loaded from session: ' + result.language); } }); }]);
